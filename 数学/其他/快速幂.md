# 快速幂

```cpp
#include <bits/stdc++.h>
using namespace std;
const int mod = 1e9 + 7;
int fastPower(int a, int k)
{
    int res = 1;
    while (k)
    {
        if (k & 1)
        {
            res = (res * a) % mod;
            k >>= 1;
            a = (long long)(a * a) % mod;
        }
    }
    return res;
}
```
