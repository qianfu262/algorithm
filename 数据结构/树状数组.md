# 树状数组

作用:快速求前缀和,快速修改某一个数
用一个tr数组来维护a数组的前缀和,add(x,c)表示a[x] += c, sum(x)表示a[1]~a[x]的和

例题:求一个数组中,每个数左边比它小的数的个数,右边比它大的数的个数

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N = 200010;
int n;
int a[N];
int tr[N];//树状数组,tr[i]表示i这个数出现的次数
int great[N], lower[N];
int lowbit(int x)
{
    return x & (-x);
}
void add(int x, int c)
{
    for (int i = x; i <= n; i += lowbit(i))
        tr[i] += c;
}
int sum(int x)//查询前缀和,x为下标
{
    int ans = 0;
    for (int i = x; i >= 1; i -= lowbit(i))
        ans += tr[i];
    return ans;
}
int main()
{
    cin >> n;
    for (int i = 1; i <= n; i++)
        cin >> a[i];
    for (int i = 1; i <= n; i++)//从左往右扫描,出现的数,其右侧的数都为大于它
    {
        int y = a[i];
        great[i] = sum(n) - sum(y);
        lower[i] = sum(y - 1);
        add(y, 1);//tr[y] += 1
    }
    memset(tr, 0, sizeof(tr));
    ll res1 = 0, res2 = 0;
    for (int i = n; i >= 1; i--)
    {
        int y = a[i];
        res1 += great[i] * (ll)(sum(n) - sum(y));
        res2 += lower[i] * (ll)(sum(y - 1));
        add(y, 1);
    }
    cout << res1 <<" "<<res2;
    return 0;
}
```
