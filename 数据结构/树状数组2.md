# 树状数组2

对于同时对一段区间进行加减操作和查询某一个数，可以使用树状数组加差分

例题:


```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N = 100010;
int n, m;
int a[N];
ll tr1[N], tr2[N];
int lowbit(int x)
{
    return x & -x;
}
void add(ll tr[], int x, ll d)
{
    for (int i = x; i <= n; i += lowbit(i))
        tr[i] += d;
}
ll sum(ll tr[], int x)
{
    ll res = 0;
    for (int i = x; i; i -= lowbit(i))
        res += tr[i];
    return res;
}
ll presum(int x)
{
    return sum(tr1, x) * (x+1) - sum(tr2, x);
}
int main()
{
    // 树状数组加差分
    cin >> n >> m;
    for (int i = 1; i <= n; i++)
        cin >> a[i];
    for (int i = 1; i <= n; i++)
    {
        int b = a[i] - a[i - 1];
        add(tr1, i, b);
        add(tr2, i, (ll)i * b);
    }
    while (m--)
    {
        char op[2];
        int l, r, d;
        cin >> op >> l >> r;
        if (op[0] == 'Q')
        {
            cout << presum(r) - presum(l - 1) << endl;
        }
        else
        {
            cin >> d;
            add(tr1, l, d);
            add(tr2, l, (ll)l * d);
            add(tr1, r + 1, -d);
            add(tr2, r + 1, (ll)(r + 1) * -d);
        }
    }
    return 0;
}
```
